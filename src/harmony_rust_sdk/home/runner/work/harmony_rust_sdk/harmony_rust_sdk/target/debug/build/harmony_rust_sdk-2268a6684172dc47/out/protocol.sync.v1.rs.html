/// Authentication data that will be sent in a `harmonytypes.v1.Token`.
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct AuthData {
    /// The server name of the server initiating the transaction. For Pull,
    /// this tells the server being connected to which homeservers' events it should send.
    /// For Push, this tells the server being connected to which homeservers' events it is
    /// receiving.
    #[prost(string, tag="1")]
    pub host: ::prost::alloc::string::String,
    /// The UTC UNIX time in seconds of when the request is started. Servers should reject
    /// tokens with a time too far from the current time, at their discretion. A recommended
    /// variance is 1 minute.
    #[prost(uint64, tag="2")]
    pub time: u64,
}
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct EventQueue {
    #[prost(message, repeated, tag="1")]
    pub events: ::prost::alloc::vec::Vec<Event>,
}
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct Event {
    #[prost(oneof="event::Kind", tags="1, 2")]
    pub kind: ::core::option::Option<event::Kind>,
}
/// Nested message and enum types in `Event`.
pub mod event {
    #[derive(Clone, PartialEq, ::prost::Message)]
    pub struct UserRemovedFromGuild {
        #[prost(uint64, tag="1")]
        pub user_id: u64,
        #[prost(uint64, tag="2")]
        pub guild_id: u64,
    }
    #[derive(Clone, PartialEq, ::prost::Message)]
    pub struct UserAddedToGuild {
        #[prost(uint64, tag="1")]
        pub user_id: u64,
        #[prost(uint64, tag="2")]
        pub guild_id: u64,
    }
    #[derive(Clone, PartialEq, ::prost::Oneof)]
    pub enum Kind {
        #[prost(message, tag="1")]
        UserRemovedFromGuild(UserRemovedFromGuild),
        #[prost(message, tag="2")]
        UserAddedToGuild(UserAddedToGuild),
    }
}
