pub struct Profile {
    /// the name of the user.
    #[prost(string, tag="1")]
    pub user_name: ::prost::alloc::string::String,
    /// the user's avatar.
    #[prost(string, optional, tag="2")]

    /// whether the user is a bot or not.
    #[prost(bool, tag="4")]
    pub is_bot: bool,
}

/// The possible statuses a user can have.
#[harmony_derive::impl_call_req(profile)]
#[harmony_derive::self_builder_with_new]
#[derive(rkyv::Archive, rkyv::Serialize, rkyv::Deserialize)]
#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord, ::prost::Enumeration)]
#[repr(i32)]
pub enum UserStatus {
    /// User is offline (not connected to the server).
    OfflineUnspecified = 0,
    /// User is online (this is the default value if ommitted).
    Online = 1,
    /// User is away.
    Idle = 2,
    /// User does not want to be disturbed.
    DoNotDisturb = 3,
    /// User is on mobile.
    Mobile = 4,
    /// User is streaming
    Streaming = 5,
}
/// Event sent when a user's profile is updated.
///
/// Servers should sent this event only to users that can "see" (eg. they are
/// in the same guild) the user this event was triggered by.

#[harmony_derive::self_builder_with_new]
#[derive(rkyv::Archive, rkyv::Serialize, rkyv::Deserialize)]
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct ProfileUpdated {
    /// User ID of the user that had it's profile updated.
    #[prost(uint64, tag="1")]
    pub user_id: u64,
    /// New username for this user.
    #[prost(string, optional, tag="2")]
    pub new_username: ::core::option::Option<::prost::alloc::string::String>,
    /// New avatar for this user.
    #[prost(string, optional, tag="3")]
    pub new_avatar: ::core::option::Option<::prost::alloc::string::String>,
    /// New status for this user.
    #[prost(enumeration="UserStatus", optional, tag="4")]
    pub new_status: ::core::option::Option<i32>,
    /// New is bot or not for this user.
    #[prost(bool, optional, tag="5")]
    pub new_is_bot: ::core::option::Option<bool>,
}
