/// AnyRequest is a generic message supporting any unary request.
#[harmony_derive::impl_call_req(batch)]
#[harmony_derive::self_builder_with_new]
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct AnyRequest {
    /// The endpoint to which the request is being sent.
    #[prost(string, tag="1")]
    pub endpoint: ::prost::alloc::string::String,
    /// The request data.
    #[prost(bytes="bytes", tag="2")]
    pub request: ::prost::bytes::Bytes,
}
/// Used in `Batch` endpoint.
#[harmony_derive::impl_call_req(batch)]
#[harmony_derive::self_builder_with_new]
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct BatchRequest {
    /// The list of requests to be executed in the batch.
    #[prost(message, repeated, tag="1")]
    pub requests: ::prost::alloc::vec::Vec<AnyRequest>,
}
/// Used in `Batch` endpoint.
#[harmony_derive::impl_call_req(batch)]
#[harmony_derive::self_builder_with_new]
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct BatchResponse {
    /// The list of responses to the requests.
    #[prost(bytes="bytes", repeated, tag="1")]
    pub responses: ::prost::alloc::vec::Vec<::prost::bytes::Bytes>,
}

#[harmony_derive::impl_call_req(batch)]
#[harmony_derive::self_builder_with_new]

    #[prost(string, tag="1")]

    /// The list of requests to be executed in the batch.
    #[prost(bytes="bytes", repeated, tag="2")]
